import time
import json

def retry(fn, *, attempts: int = 3, base_delay: float = 0.5):
    last_exception = None
    for attempt in range(attempts):
        try:
            result = fn()
            # If the function call is successful, return the result as JSON
            return json.dumps({"code": "success", "explanation": result})
        except Exception as e:
            last_exception = e
            # Calculate the backoff delay
            delay = base_delay * (2 ** attempt)
            time.sleep(delay)
    
    # If we exhausted all attempts, re-raise the last exception
    raise last_exception
